# 변경사항

- 멀티 프로세싱 지원
- 쓰레드 및 프로세스 초기화 지원
- 이벤트 러너 초기화 지원

## 사용방법

```python
# -*- coding: utf-8 -*-

from Liquirizia.EventWorker import (
	Worker,
	Invoker,
	Pool,
	Setup as BaseSetup,
	ThreadPool,
	ProcessPool,
)
from Liquirizia.EventWorker.Tools.StatusChecker import StatusChecker
from Liquirizia.EventWorker import (
	EventRunner,
	EventParameters,
	EventSetup,
	EventComplete,
	EventError,
	EventProperties,
	EventContext,
	Factory,
)

from multiprocessing import get_context
from signal import signal, SIGINT
from random import randint
from time import sleep


class SampleInvoker(Invoker):
	def __init__(self, pool: Pool):
		self.event = get_context().Manager().Event()
		self.pool = pool
		self.context = EventContext()
		return
	def run(self):
		EVENT = ['+', '-', '*', '/', '%']
		while not self.event.is_set():
			event = EVENT[randint(0, 4)]
			a = randint(0, 9)
			b = randint(0, 9)
			LOG_INFO('Event : type={}, {}'.format(event, {'a': a, 'b': b}))
			tid = self.pool.run(
				event=event,
				parameters=EventParameters(**{'a': a, 'b': b}),
			)
			if not tid:
				LOG_ERROR('Event run failed : type={}, {}'.format(event, {'a': a, 'b': b}))
			sleep(1)
		return
	def stop(self):
		self.event.set()
		return


class Init(BaseSetup):
	def __call__(self):
		print('Process Initialized')
		return


class Setup(EventSetup):
	def __call__(self, parameters: EventParameters):
		print('Setup : parameters={}'.format(parameters))
		return


class Complete(EventComplete):
	def __call__(self, parameters: EventParameters, completion):
		print('Complete : parameters={}, completion={}'.format(parameters, completion))
		return


class Error(EventError):
	def __call__(self, parameters: EventParameters, error: BaseException):
		print('Error : parameters={}, error={}'.format(parameters, str(error)), e=error)
		return


@EventProperties(
	event='+',
	setups=Setup(),
	completes=Complete(),
	errors=Error(),
)
class Add(EventRunner):
	def run(self, a: int , b: int):
		sleep(randint(1, 5))
		return a + b

@EventProperties(
	event='-',
	setups=Setup(),
	completes=Complete(),
	errors=Error(),
)
class Sub(EventRunner):
	def run(self, a: int , b: int):
		sleep(randint(1, 5))
		return a - b

@EventProperties(
	event='*',
	setups=Setup(),
	completes=Complete(),
	errors=Error(),
)
class Mul(EventRunner):
	def run(self, a: int , b: int):
		sleep(randint(1, 5))
		return a * b

@EventProperties(
	event='/',
	setups=Setup(),
	completes=Complete(),
	errors=Error(),
)
class Div(EventRunner):
	def run(self, a: int , b: int):
		sleep(randint(1, 5))
		return a / b


@EventProperties(
	event='%',
	setups=Setup(),
	completes=Complete(),
	errors=Error(),
)
class Mod(EventRunner):
	def run(self, a: int , b: int):
		sleep(randint(1, 5))
		return a % b

if __name__ == '__main__':
	LOG_INIT(LOG_LEVEL_DEBUG)
	sc = StatusChecker()
	worker = Worker(ProcessPool(setups=Init()), SampleInvoker)
	def stop(signal, frame):
		worker.stop()
		sc.stop()
		return
	signal(SIGINT, stop)
	sc.start()
	worker.run()
```
